\section{Motivation and Problem Statements}

\begin{frame}{Motivation}
  \begin{itemize}
    \item Citation networks: nodes are papers, directed edges are citations
    \item Query: a paper $q$
    \item Goal: retrieve a compact, coherent citation community around $q$
    \item Baseline: take immediate citation neighbors
    \item Issue: immediate neighbors are not an optimization problem
  \end{itemize}

  \vfill

  \textbf{Question} Can we search for a neighborhood that is dense, query-centered, and structurally constrained?
\end{frame}

\begin{frame}{What makes this hard?}
  \begin{itemize}
    \item The objective is fractional:
    \[
      \frac{|E(S)|}{|S|(|S|-1)}
    \]
    \item The decision variables are binary: select papers or not
    \item The graph can be huge, or only available through an oracle
    \item Connectivity constraints are global
    \item Exact search can be expensive even on moderate local graphs
  \end{itemize}

  \vfill

  \textbf{Design choice} Use exact optimization machinery, but expose time limits and bounds.
\end{frame}

\begin{frame}{Combinatorial problem}
  Given
  \begin{itemize}
    \item directed graph $G=(V,E)$
    \item query node $q \in V$
    \item minimum size $k \ge 3$
  \end{itemize}

  \vfill

  Solve
  \begin{align*}
    \max_{S \subseteq V} \quad &
      \frac{|E(S)|}{|S|(|S|-1)} \\
    \mathrm{s.t.} \quad &
      q \in S \\
    & |S| \ge k
  \end{align*}

  \vfill

  where $E(S)=\{(u,v)\in E \mid u,v\in S\}$.
\end{frame}

\begin{frame}{Optional connectivity}
  \textbf{Rooted $\kappa$-edge-connectivity}

  \vfill

  For every selected node $v \in S \setminus \{q\}$:
  \[
    \text{there are at least } \kappa
    \text{ edge-disjoint paths from } q \text{ to } v.
  \]

  \vfill

  \begin{itemize}
    \item Checked in the undirected support of selected citation edges
    \item Enforced lazily by min-cut inequalities
    \item $\kappa=0$ disables the constraint
    \item $\kappa=1$ was selected by the Cora-ML BP validation sweep
  \end{itemize}
\end{frame}

\begin{frame}{Nonlinear fractional 0-1 program}
  Let
  \begin{itemize}
    \item $x_v \in \{0,1\}$ indicate $v \in S$
    \item $y_{uv} \in \{0,1\}$ indicate directed edge $(u,v)$ is internal
  \end{itemize}

  \begin{align*}
    \max \quad &
      \frac{\sum_{(u,v)\in E} y_{uv}}
      {\left(\sum_{u\in V}x_u\right)
       \left(\sum_{u\in V}x_u - 1\right)}
       \\
    \mathrm{s.t.} \quad &
      x_q = 1 \\
    & \sum_{u\in V} x_u \ge k \\
    & y_{uv} \le x_u,\quad y_{uv}\le x_v
      \quad \forall (u,v)\in E
  \end{align*}

  \vfill

  \textbf{Note} The lower bound $y_{uv}\ge x_u+x_v-1$ is not needed because $y$ has positive objective coefficient.
\end{frame}

\section{Approaches}

\begin{frame}{Dinkelbach transform}
  For a density estimate $\lambda$, define
  \[
    Z(\lambda)
    =
    \max_S
    \left\{
      |E(S)| - \lambda |S|(|S|-1)
    \right\}.
  \]

  \vfill

  \textbf{Key property}
  \[
    \lambda^\star \text{ is optimal}
    \quad \Longleftrightarrow \quad
    Z(\lambda^\star)=0.
  \]

  \vfill

  \begin{itemize}
    \item If $Z(\lambda)>0$, an improving denser set exists
    \item Update $\lambda \gets |E(S)|/[|S|(|S|-1)]$
    \item Stop when the parametric subproblem cannot improve density
  \end{itemize}
\end{frame}

\begin{frame}{Dinkelbach loop}
  \begin{enumerate}
    \item Build an initial feasible set by BFS from $q$
    \item Prune it to improve density
    \item Set $\lambda$ to the current density
    \item Solve the parametric integer subproblem
    \item If objective $\le 0$, stop
    \item Otherwise update $\lambda$ and repeat
  \end{enumerate}

  \vfill

  \textbf{Outer loop} handles the fractional objective.

  \vfill

  \textbf{Inner loop} solves the parametric problem by branch-and-price.
\end{frame}

\begin{frame}{Linearizing the penalty}
  The Dinkelbach penalty is
  \[
    |S|(|S|-1)
    =
    \left(\sum_v x_v\right)^2 - \sum_v x_v.
  \]

  Since $x_v^2=x_v$ for binary variables:
  \[
    |S|(|S|-1)
    =
    2\sum_{u<v}x_u x_v.
  \]

  \vfill

  Introduce pair variables:
  \[
    w_{uv} \approx x_u x_v,
    \qquad
    w_{uv} \ge x_u+x_v-1.
  \]

  \vfill

  Because $w$ has negative coefficient, this lower bound is enough at integer solutions.
\end{frame}

\begin{frame}{Oracle model}
  The solver does not need the whole graph.

  \vfill

  \begin{itemize}
    \item \textbf{Simulation oracle:} read local CSV edge list
    \item \textbf{OpenAlex oracle:} fetch predecessors and successors live
    \item Each query reveals one node's 1-hop citation neighborhood
  \end{itemize}

  \vfill

  Maintained sets:
  \begin{itemize}
    \item $A$: active nodes in the restricted master problem
    \item $F$: frontier nodes discovered but not active
    \item cached in/out adjacency lists
    \item pending edges whose endpoint variables are not both created yet
  \end{itemize}
\end{frame}

\begin{frame}{Directionality and max-in-degree cap}
  Edge convention:
  \[
    u \to v
    \quad \Longleftrightarrow \quad
    \text{paper } u \text{ cites paper } v.
  \]

  \vfill

  Oracle query for node $v$:
  \begin{itemize}
    \item outgoing neighbors: papers referenced by $v$
    \item incoming neighbors: papers that cite $v$
  \end{itemize}

  \vfill

  \textbf{\texttt{max\_in\_edges}}
  \begin{itemize}
    \item caps incoming/cited-by edges fetched per queried node
    \item prevents explosion around highly cited papers
    \item reported Cora-ML runs use \texttt{max\_in\_edges = 0}
    \item therefore: no incoming expansion, outgoing citation direction only
  \end{itemize}
\end{frame}

\begin{frame}{Active set initialization}
  \begin{itemize}
    \item Start from query $q$
    \item BFS outward until $|A|\ge k$
    \item Query any active node that was added as a neighbor but not expanded
    \item Add unseen adjacent nodes to frontier $F$
    \item Blacklist nodes that fail oracle retrieval
  \end{itemize}

  \vfill

  \textbf{Purpose}
  \begin{itemize}
    \item guarantee a feasible initial model
    \item avoid loading the full citation graph
    \item seed Dinkelbach with a meaningful local density
  \end{itemize}
\end{frame}

\begin{frame}{Restricted Master Problem}
  For active set $A$:

  \begin{align*}
    \max \quad &
      \sum_{(u,v)\in E(A)} y_{uv}
      -2\lambda\sum_{\substack{u,v\in A\\u<v}}w_{uv} \\
    \mathrm{s.t.}\quad &
      x_q = 1 \\
    & \sum_{u\in A}x_u \ge k
      \quad (\text{dual: }\pi) \\
    & y_{uv}\le x_u,\quad y_{uv}\le x_v
      \quad \forall (u,v)\in E(A) \\
    & w_{uv}\ge x_u+x_v-1
      \quad \forall u<v \in A \\
    & 0\le x,y,w\le 1
  \end{align*}

  \vfill

  This LP gives the bound inside branch-and-bound.
\end{frame}

\begin{frame}{Pricing a frontier node}
  Let $\bar{x}$ be the RMP solution.

  \vfill

  Fractional degree of a frontier node:
  \[
    \mathrm{fracdeg}(f)
    =
    \sum_{v\in N(f)\cap A}\bar{x}_v.
  \]

  Reduced cost score:
  \[
    \bar{c}_f
    =
    \mathrm{fracdeg}(f)
    -2\lambda\sum_{v\in A}\bar{x}_v
    -\pi.
  \]

  \vfill

  \begin{itemize}
    \item edge gain: $\mathrm{fracdeg}(f)$
    \item size-pair penalty: $-2\lambda\sum_v\bar{x}_v$
    \item size dual adjustment: $-\pi$
  \end{itemize}
\end{frame}

\begin{frame}{Column generation}
  \textbf{Pricing rule}
  \[
    f^\star = \argmax_{f\in F}\bar{c}_f.
  \]

  \vfill

  \begin{itemize}
    \item If $\bar{c}_{f^\star}\le 0$, no frontier node improves the LP
    \item If $\bar{c}_{f^\star}>0$, promote frontier nodes into $A$
    \item The implementation adds a batch, not just one node
  \end{itemize}

  \vfill

  Batch size:
  \[
    b =
    \mathrm{clamp}
    \left(
      \lfloor \rho |A| \rfloor,\ b_{\min},\ b_{\max}
    \right).
  \]

  \vfill

  \textbf{Interpretation} Column generation is also graph expansion.
\end{frame}

\begin{frame}{Branch-and-bound}
  Each B\&B node stores:
  \begin{itemize}
    \item $V_1$: nodes fixed into the solution
    \item $V_0$: nodes fixed out of the solution
  \end{itemize}

  \vfill

  At each node:
  \begin{enumerate}
    \item apply bounds to the persistent RMP
    \item run column generation
    \item prune if LP bound cannot beat incumbent
    \item branch if $\bar{x}$ is fractional
    \item check integer solution if $\bar{x}$ is integral
  \end{enumerate}

  \vfill

  Root node fixes $q$ into the solution.
\end{frame}

\begin{frame}{Branching rule}
  A fractional node is judged by its fractional internal degree.

  \vfill

  \[
    d_{\mathrm{frac}}(v)
    =
    \sum_{u\in N(v)\cap A}\bar{x}_u.
  \]

  \vfill

  \begin{itemize}
    \item If $d_{\mathrm{frac}}(v) < 2\lambda$:
    \begin{itemize}
      \item $v$ is a hanging node
      \item branch zero-first
    \end{itemize}
    \item Otherwise:
    \begin{itemize}
      \item $v$ is more core-like
      \item branch one-first
    \end{itemize}
  \end{itemize}

  \vfill

  \textbf{Reason} $2\lambda$ is the marginal pair penalty threshold.
\end{frame}

\begin{frame}{Triangle cuts}
  LP relaxation can spread mass across nodes while keeping pair variables small.

  \vfill

  Add Boolean quadratic polytope cuts:
  \[
    x_u+x_v+x_w
    -w_{uv}-w_{vw}-w_{uw}
    \le 1.
  \]

  \vfill

  \begin{itemize}
    \item valid for binary $x$ and $w=x_ix_j$
    \item separated only on fractional nodes
    \item cap: 20 violated cuts per round
    \item skipped when LP gap is already small
  \end{itemize}

  \vfill

  \textbf{Goal} tighten the bound without overwhelming the RMP.
\end{frame}

\begin{frame}{$\kappa$-connectivity cuts}
  For an integer candidate:
  \begin{enumerate}
    \item Build unit-capacity support graph on selected edges
    \item For each $t\in S\setminus\{q\}$, compute max flow $q\to t$
    \item If flow $<\kappa$, extract min-cut $R$
    \item Add cut inequality
  \end{enumerate}

  \vfill

  \[
    \sum_{\{u,v\}\in \delta(R)} y_{uv}
    \ge
    \kappa x_t.
  \]

  \vfill

  \textbf{Lazy enforcement} Only add connectivity constraints when a candidate violates them.
\end{frame}

\begin{frame}{Pruning integer incumbents}
  Since the constraint is $|S|\ge k$, solutions may be larger than needed.

  \vfill

  Greedy pruning:
  \begin{itemize}
    \item never remove query node $q$
    \item compute internal degree of each selected node
    \item remove the least useful node if objective improves
    \item stop at size $k$ or when no removal helps
    \item preserve connectivity when $\kappa>0$
  \end{itemize}

  \vfill

  \textbf{Effect} stronger incumbent, smaller returned community.
\end{frame}

\begin{frame}{Average-degree baseline: pipeline}
  \textbf{Problem.} Maximise $|E(S)| / |S|$ over $S \ni q$,
  $S \subseteq V_{\mathrm{loc}}$ where $V_{\mathrm{loc}}$ is BFS-bounded.
  No size constraint, no connectivity constraint, no triangle
  relaxation. Different objective from BP: avg degree, not pair
  density.

  \vfill

  \textbf{Pipeline.}
  \begin{itemize}
    \item BFS from $q$ caches the local subgraph once.
    \item Dinkelbach bisects $\lambda \in [0, |E_{\mathrm{loc}}|]$.
    \item Per $\lambda$: Goldberg-style $s$--$t$ flow network, then
    push-relabel max-flow. The source-side residual gives $S_\lambda$.
    \item Keep the argmax density across every probed $\lambda$, not
    only the converged one.
  \end{itemize}
\end{frame}

\begin{frame}{Average-degree baseline: four adaptations of Goldberg 1984}
  \begin{itemize}
    \item \emph{Query anchor.} $s \to q$ capacity $\infty$ keeps $q$ on
    the source side of every cut, so the answer is query-anchored
    densest.
    \item \emph{Local subgraph.} BFS bound replaces Goldberg's
    fully-materialised graph; required for the lazy oracle.
    \item \emph{Directed unit-weight edges.} Each directed edge
    $(u, v)$ contributes one edge-item, so density counts directed
    edges; a reciprocal pair $(u, v), (v, u)$ contributes two.
    \item \emph{Objective.} Avg degree, not pair density. The two
    objectives are not monotone in each other.
  \end{itemize}
\end{frame}

\section{Experiments}

\begin{frame}{Exp 1: solver correctness on enumerable graphs}
  \textbf{Goal.} Verify that the C++ branch-and-price binary recovers
  the true edge-density and avg-degree optima on graphs small enough
  to enumerate exhaustively.

  \vfill

  \textbf{Fixture.} Erd\H{o}s--R\'enyi random graphs $G(n, p)$ at
  $n = 20$, $p \in \{0.25,\, 0.50,\, 0.75\}$, seeds $0..4$. Fifteen
  graphs total. Query node $q$ is the highest-degree node per graph.
\end{frame}

\begin{frame}{Exp 1: brute-force oracle}
  \textbf{Enumerator.} For every $S \ni q$ with $|S| \ge k$:
  \begin{itemize}
    \item Iterate $k \in \{3, 4, 5\}$ and
    $\kappa \in \{0, 1, 2\}$.
    \item Record the maximum-density $S$ subject to
    $\lambda(S) \ge \kappa$.
    \item Cells with no feasible $S$ at the requested $\kappa$ store a
    $-1$ sentinel in \texttt{optima.csv}.
  \end{itemize}

  \vfill

  Every row carries a \texttt{code\_hash} (sha256 of the brute-force
  script) so a stale CSV is rejected at aggregation.
\end{frame}

\begin{frame}{Exp 1: solver protocol}
  \textbf{Runs.} The C++ binary on every
  $(\mathrm{method}, k, \kappa, \mathrm{seed})$ cell with no time,
  node, gap, or Dinkelbach limit.

  \vfill

  \textbf{Skip rule.} Cells brute-force-infeasible (the $-1$ sentinel
  in \texttt{optima.csv}) are not dispatched; without this the solver
  exhausts B\&B trying to prove infeasibility and wall time blows up
  on sparse seeds.

  \vfill

  Every row in \texttt{solver\_runs.csv} carries the matching
  \texttt{solver\_build\_id} (git short SHA + UTC build time baked at
  \texttt{cmake} time) so a stale binary is rejected at aggregation.
\end{frame}

\begin{frame}{Exp 1: result}
  \textbf{AvgDeg} matches the unconstrained avg-degree optimum on
  $15 / 15$ graphs at every $p$.

  \vfill

  \textbf{BP} matches the edge-density optimum on every feasible
  $(k, \kappa, p)$ cell. Two cells at $k = 3$, $\kappa \in \{1, 2\}$,
  $p = 0.5$ have one seed where no $S$ reaches the requested $\kappa$;
  that seed is surfaced as \texttt{infeasible\_kappa} rather than
  counted as a non-match.
\end{frame}

\begin{frame}{Exp 3: classification dataset and split}
  Dataset: CitationFull Cora-ML (lead dataset).

  \vfill

  \begin{itemize}
    \item 2,995 nodes, 8,416 directed edges, 7 labels.
    \item Pool: pure-source papers (citing-only) with out-degree $\ge 2$.
    \item Pool split 50 / 50: val = 402, test = 402 (seed 42).
    \item Train = every other node (2,191 on Cora\_ML).
    \item Hard subset: 64 val nodes where BFS depth-1 label vote is wrong.
  \end{itemize}

  \vfill

  All splits, train mask, and hard subset are hash-pinned in
  \texttt{data/Cora\_ML/split\_meta.json} (schema $2.0$,
  \texttt{pool\_criterion = pure\_source\_with\_outdegree\_geq\_2}).
\end{frame}

\begin{frame}{Exp 3: classification prediction rule}
  For each val or test query $q$:
  \begin{itemize}
    \item Retrieve the community around $q$ from the chosen solver.
    \item Uniform majority over the train-labelled nodes in the
    community.
    \item Ties broken deterministically by ascending label value
    (\texttt{argmax\_label}).
    \item If no train neighbour is present in the community, fall back
    to a BFS walk that \emph{forbids} every val and test node as a
    stepping stone (no label leakage across splits).
    \item Last resort if BFS reaches no train label: global majority of
    the train pool.
  \end{itemize}

  \vfill

  All Cora\_ML runs use \texttt{max\_in\_edges = 0}.
\end{frame}

\begin{frame}{Methods compared}
  \begin{itemize}
    \item \textbf{BFS depth 1}
    \begin{itemize}
      \item immediate outgoing citation neighborhood
      \item cheapest baseline
    \end{itemize}
    \item \textbf{BFS depth 2/3}
    \begin{itemize}
      \item broader local context in validation sweep
      \item larger neighborhoods
    \end{itemize}
    \item \textbf{BP}
    \begin{itemize}
      \item branch-and-price for directed density
      \item validation sweep over $k$ and $\kappa$
    \end{itemize}
    \item \textbf{Avgdeg}
    \begin{itemize}
      \item exact average-degree baseline on local BFS graph
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Why look at a hard subset?}
  The full test split mixes two regimes.

  \vfill

  \begin{itemize}
    \item Easy regime:
    \begin{itemize}
      \item one-hop citation labels are already correct
      \item BFS depth 1 is hard to beat
    \end{itemize}
    \item Hard regime:
    \begin{itemize}
      \item one-hop BFS predicts the wrong label
      \item local labels are misleading
      \item retrieval geometry has to compensate
    \end{itemize}
  \end{itemize}

  \vfill

  \textbf{BFS-hard validation subset:} 52 / 411 eligible val nodes where
  the BFS depth 1 vote disagrees with the ground-truth label. The subset is
  hash-pinned in \texttt{split\_meta.json}.
\end{frame}

\section{Conclusion}

\begin{frame}{Main interpretation}
  \textbf{Cluster quality is the headline; classification is the supporting signal.}

  \vfill

  \begin{itemize}
    \item BP returns the densest, best-mixed subgraph of the three methods
      on Cora\_ML, at a community size close to BFS depth 1.
    \item BFS depth 1 wins macro F1 on the full test split because Cora\_ML
      test nodes cite training papers of the same class.
    \item On the BFS-hard subset, the spectral and density picture stays the
      same: BFS still has the highest $\lambda_2$, BP returns the densest
      community, AvgDeg spreads out far past the local cluster.
    \item Classification ranks reorder on the hard subset: BFS collapses to 0
      by construction, and AvgDeg and BP both move off zero.
  \end{itemize}

  \vfill

  \textbf{Claim:} BP is a principled dense-community retriever. Its
  competitive classification F1 follows from the geometry, not the other
  way round.
\end{frame}

\begin{frame}{Limitations}
  \begin{itemize}
    \item Reported numbers are Cora\_ML only. CiteSeer, DBLP, and Cora runs
      are in progress.
    \item The temporal split makes one-hop BFS unusually strong: test nodes
      are pure sources whose direct citations are in the train set.
    \item The hard subset is hash-pinned but defined on validation; it is a
      diagnostic, not a held-out test.
    \item BP relies on Gurobi.
    \item Exact global optimality is over the discovered graph region; the
      oracle interface limits what BP can reason about.
    \item OpenAlex mode depends on API coverage and latency.
  \end{itemize}
\end{frame}

\begin{frame}{Next steps}
  \begin{itemize}
    \item Multi-dataset classification experiments
    \item Synthetic planted-community benchmarks
    \item Ablate:
    \begin{itemize}
      \item $\kappa$ cuts

    \end{itemize}
    \item Add latency-aware OpenAlex experiments
    \item Explore hybrid objectives:

    \]
  \end{itemize}
\end{frame}

\begin{frame}{Takeaways}
  \begin{itemize}
    \item Query-anchored dense community search is a different task from BFS classification
    \item Dinkelbach handles the fractional density objective
    \item Branch-and-price handles binary search over a lazy active graph
    \item Pricing decides which frontier papers to fetch
    \item Connectivity can be enforced lazily by min-cut cuts
    \item Results show a two-regime story:
    \begin{itemize}
      \item BFS wins easy local-label cases
      \item BP is useful for compact constrained retrieval
    \end{itemize}
  \end{itemize}
\end{frame}
