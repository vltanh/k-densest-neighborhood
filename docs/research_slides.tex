\section{Motivation and problem statements}

\begin{frame}{Motivation}
  \begin{itemize}
    \item Citation networks: nodes are papers, directed edges are citations
    \item Query: a paper $q$
    \item Goal: retrieve a ``good'' citation community around $q$
  \end{itemize}

  \vfill

  \textbf{Question.} Can we search for a neighborhood that is dense, query-centered, and structurally constrained?
\end{frame}

\begin{frame}{Oracle access}
  Edge convention: $u \to v \;\Longleftrightarrow\; u$ cites $v$.

  \vfill

  The graph is available only through 1-hop neighborhood queries. Each query for a paper $v$ returns its outgoing references and its incoming citations.

  \vfill

  \textbf{Incoming-citation cap}
  \begin{itemize}
    \item incoming-edge queries can be more expensive than outgoing:
          \begin{itemize}
            \item reference list is usually small, cited-by list is
                  unlimited
            \item references read off the paper itself; cited-by needs a
                  maintained database
          \end{itemize}
    \item the cap limits how many incoming edges are fetched per
          queried node
  \end{itemize}
\end{frame}

\begin{frame}{Combinatorial problem}
  Given
  \begin{itemize}
    \item directed graph $G=(V,E)$
    \item query node $q \in V$
    \item minimum size $k \ge 3$
  \end{itemize}

  \vfill

  Solve
  \begin{align*}
    \max_{S \subseteq V} \quad &
    \frac{|E(S)|}{|S|(|S|-1)}              \\
    \mathrm{s.t.} \quad        &
    q \in S                                \\
                               & |S| \ge k
  \end{align*}

  \vfill

  where $E(S)=\{(u,v)\in E \mid u,v\in S\}$.
\end{frame}

\begin{frame}{Optional connectivity}
  \textbf{Rooted $\kappa$-edge-connectivity}

  \vfill

  For every selected node $v \in S \setminus \{q\}$:
  \[
    \text{there are at least } \kappa
    \text{ edge-disjoint paths from } q \text{ to } v
  \]
  in the underlying undirected graph.

  \vfill

  \textbf{Note}
  \begin{itemize}
    \item $\kappa = 0$ disables the constraint.
    \item Enforced lazily by min-cut cuts inside the solver.
  \end{itemize}
\end{frame}

\section{Approaches}

\begin{frame}{Dinkelbach transform}
  For a density estimate $\lambda$, define
  \[
    Z(\lambda)
    =
    \max_S
    \left\{
    |E(S)| - \lambda |S|(|S|-1)
    \right\}.
  \]

  \vfill

  \textbf{Property}
  \[
    \lambda^\star \text{ is optimal}
    \quad \Longleftrightarrow \quad
    Z(\lambda^\star)=0.
  \]

  \vfill

  \begin{itemize}
    \item If $Z(\lambda)>0$, an improving denser set exists
    \item Update $\lambda \gets |E(S)|/[|S|(|S|-1)]$
    \item Stop when the parametric subproblem cannot improve density
  \end{itemize}
\end{frame}

\begin{frame}{Dinkelbach loop}
  \begin{enumerate}
    \item \textbf{Initialize:} build a feasible set by BFS from $q$, prune to improve density, set $\lambda_1$ to its density, $t = 1$
    \item \textbf{Optimize:} solve the parametric integer subproblem at $\lambda_t$
    \item \textbf{Convergence check:} if $Z(\lambda_t) \le 0$, terminate and return $\lambda_t$
    \item \textbf{Update:} otherwise set $\lambda_{t+1}$ to the density of the new subgraph, $t \gets t+1$, return to step 2
  \end{enumerate}

  \vfill

  \textbf{Outer loop} handles the fractional objective.

  \vfill

  \textbf{Inner loop} solves the parametric problem by branch-and-price.

  \vfill

  \footnotesize See appendix A5 for the LP formulation and linearisation.
\end{frame}

\begin{frame}{Active set and frontier}
  \textbf{Maintained sets}
  \begin{itemize}
    \item $A$: active nodes in the restricted master problem
    \item $F$: frontier nodes discovered but not active
    \item subgraph induced by $A \cup F$ is kept up to date via oracle queries, so no internal edge is missed
  \end{itemize}

  \vfill

  \textbf{Initialization}
  \begin{itemize}
    \item BFS rooted at $q$: each dequeued node is oracle-queried, joins $A$, and its neighbors enter $F$
    \item loop terminates at $|A| = k$
  \end{itemize}
\end{frame}

\begin{frame}{Column generation}
  Reduced cost of a frontier node $f \in F$ at RMP optimum $\bar{x}$ with size-constraint dual $\pi$:
  \[
    \bar{c}_f
    = \underbrace{d_{\mathrm{frac}}(f)}_{\text{edge gain}}
    \;-\; \underbrace{2\lambda \sum_{v \in A} \bar{x}_v}_{\text{pair penalty}}
    \;-\; \underbrace{\pi}_{\text{size dual}},
    \qquad d_{\mathrm{frac}}(f) = \sum_{v \in N(f) \cap A} \bar{x}_v.
  \]

  Pick $f^\star = \argmax_{f \in F} \bar{c}_f$. If $\bar{c}_{f^\star} \le 0$, RMP is a valid upper bound and CG terminates. Otherwise promote the top-$b$ frontier nodes (batched) into $A$ and resolve.

  \vfill

  \footnotesize See appendix A1 (fractional degree) and A2 (reduced cost derivation).
\end{frame}

\begin{frame}{Branch-and-bound}
  Each B\&B node stores $V_1$ (fixed in) and $V_0$ (fixed out). At each node: apply bounds, run column generation, prune if LP bound cannot beat incumbent. If $\bar{x}$ integral, accept as a candidate; else branch.

  \vfill

  \textbf{Branching rule.} Classify a fractional node by $d_{\mathrm{frac}}(v) = \sum_{u \in N(v) \cap A} \bar{x}_u$ versus $2\lambda$.
  \begin{itemize}
    \item $d_{\mathrm{frac}}(v) < 2\lambda$: "hanging", branch zero-first.
    \item $d_{\mathrm{frac}}(v) \ge 2\lambda$: "core", branch one-first.
  \end{itemize}
  $2\lambda$ is the marginal pair-penalty rate. Root fixes $x_q = 1$.

  \vfill

  \footnotesize See appendix A3.
\end{frame}

\begin{frame}{Triangle cuts}
  \textbf{Problem.} LP relaxation can spread fractional $x$ mass across nodes while keeping pair variables $w_{ij}$ at zero (e.g., $x_u = x_v = 0.5$ and $w_{uv} = 0$).

  \vfill

  \textbf{Solution.} Add constraints that forbid this pattern:
  \[
    x_u + x_v + x_w - w_{uv} - w_{vw} - w_{uw} \le 1.
  \]

  \vfill

  \textbf{Note}
  \begin{itemize}
    \item separated only on fractional nodes
    \item bounded number per round
    \item skipped if LP gap is small
  \end{itemize}
\end{frame}

\begin{frame}{$\kappa$-connectivity cuts}
  Lazy enforcement. On each integer candidate $S$, scan $t \in S \setminus \{q\}$ via $q \to t$ max-flow on the unit-capacity undirected support graph.

  \vfill

  At the first $t$ with flow $< \kappa$:
  \begin{itemize}
    \item $R$ = source-side of min-cut (residual-reachable from $q$)
    \item add cut inequality
          \[
            \sum_{(u,v) \in \delta(R)} y_{uv} \;\ge\; \kappa \, x_t
          \]
          where $\delta(R) = \{(u,v) \in E : \text{exactly one of } u, v \in R\}$
    \item re-solve
  \end{itemize}

  \vfill

  One cut per round; the next violation, if any, is caught on re-solve.

  \vfill

  \footnotesize See appendix A4.
\end{frame}

\begin{frame}{Practical considerations}
  \textbf{Early stopping}
  \begin{itemize}
    \item soft time limit: no-improvement cap, fires only when an incumbent exists
    \item hard time limit: absolute wall-clock cap, fires regardless
    \item node limit: cap branch-and-bound nodes per subproblem
    \item gap tolerance: terminate when LP-incumbent gap drops below threshold
  \end{itemize}

  \vfill

  \textbf{Lazy frontier materialization}
  \begin{itemize}
    \item fetch unqueried frontier adjacencies only when solo pricing finds no improving column
  \end{itemize}

  \vfill

  With any limit active, the solver returns the best incumbent. The LP bound stays a valid upper envelope on the parametric optimum.
\end{frame}

\begin{frame}{BFS baseline}
  \textbf{Problem.} Topology-only frontier expansion from $q$.

  \vfill

  \textbf{Algorithm.} Layered breadth-first expansion from the query: each dequeued node is oracle-queried, its predecessors and successors enter the pool. Depth $d$ bounds the number of layers explored.

  \vfill

  \textbf{vs BP}
  \begin{itemize}
    \item no density objective: returns the whole frontier
    \item no size constraint $|S| \ge k$
    \item no $\kappa$-connectivity constraint
  \end{itemize}

  \vfill

  \footnotesize Cheapest baseline: one oracle query per visited node, no flow, no LP. Output size grows with $d$; comparable per-$k$ analysis needs the grow-to-$k$ post-processing.
\end{frame}

\begin{frame}{Average-degree baseline}
  \textbf{Problem}
  \[
    \max_{\substack{S \ni q \\ S \subseteq V_{\mathrm{loc}}}} \frac{|E(S)|}{|S|}
  \]
  where $V_{\mathrm{loc}}$ is a BFS-bounded local subgraph around $q$.

  \vfill

  \textbf{vs BP}
  \begin{itemize}
    \item objective: avg degree, not pair density (the two are not monotone in each other)
    \item no size constraint $|S| \ge k$
    \item no $\kappa$-connectivity constraint
  \end{itemize}

  \vfill

  \footnotesize Algorithm: adapted Goldberg's $s$--$t$ max-flow algorithm (polynomial time)
  \begin{itemize}
    \item materializing the local subgraph up-front
    \item forcing $q \in S$ by keeping $q$ on the source side of the cut
  \end{itemize}
\end{frame}

\begin{frame}{Grow-to-$k$}
  Equalizes baseline output cardinality with BP for per-$k$ comparison.

  \vfill

  \textbf{Rule.} Seed $S$, pool $P \supseteq S$. Repeat
  \[
    v^* = \arg\max_{v \in P \setminus S} \bigl|N(v) \cap S\bigr|
  \]
  until $|S| = k$ (or $P$ is exhausted).

  \vfill

  \begin{itemize}
    \item BFS: seed $= \{q\}$, depth $d$ becomes a minimum; expansion continues until the pool reaches $k$.
    \item AvgDeg: seed $= S^*$, fires only when $|S^*| < k$.
  \end{itemize}

  \vfill

  \footnotesize Greedy heuristic. Densest-at-least-$k$ is NP-hard; no approximation guarantee.
\end{frame}

\section{Experiments}

\begin{frame}{Exp 1: solver correctness on enumerable graphs}
  Erd\H{o}s--R\'enyi $G(n=20, p)$ for $p \in \{0.25, 0.50, 0.75\}$, seeds $0..4$ (15 graphs). Brute-force enumerates every $S \ni q$.

  \vfill

  \textbf{Result on the grid $(k, \kappa) \in \{3,4,5\} \times \{0,1,2\}$}
  \begin{itemize}
    \item AvgDeg matches BF optimum on $15/15$ graphs.
    \item BP returned $S$ satisfies $|S| \ge k$ and rooted $\kappa$-edge-connectivity on $30/30$ cells; edge density matches BF on every feasible cell.
    \item Zero solver errors.
  \end{itemize}

  \vfill

  \footnotesize See appendix A6 for protocol details.
\end{frame}

  \begin{itemize}
    \item BFS depth-$1$, AvgDeg (Goldberg), BP
    \item all grown to $|S| \ge k$ (unless not possible)

  \end{itemize}

\begin{frame}{Exp 3: scope and prediction rule}
  \begin{itemize}
    \item extrinsic perspective
    \item tune on val, evaluate on test and on the BFS-hard test
          subset
  \end{itemize}

  \vfill

  For each val or test query $q$:
  \begin{itemize}
    \item retrieve a community around $q$ from the chosen method
    \item predict $q$'s label by a uniform majority vote over the training-labelled members of that community
    \item ties broken by ascending label so the result is
          reproducible
    \item fallback (rare):
          \begin{itemize}
            \item walk outward by BFS from $q$, forbidding val and test nodes as stepping stones, until a training neighbour is reached
            \item if no training label is reachable, use the global training-majority label
          \end{itemize}
  \end{itemize}
\end{frame}

% ── A1. Fractional degree (refs: Pricing a frontier node) ────────────────────

\begin{frame}{A1. Fractional degree: directionality and cap}
  \textbf{Directionality}
  \begin{itemize}
    \item $N(v) = N^+(v) \cup N^-(v)$: undirected union of outgoing references and incoming citations
    \item reciprocal pairs $(u,v),(v,u) \in E$ contribute $v$ twice, matching directed edge density
  \end{itemize}

  \vfill

  \textbf{Robustness to the incoming-citation cap}
  \begin{itemize}
    \item every active node is oracle-queried, so its complete outgoing list is recorded
    \item ingestion is symmetric: writing $u \to v$ also marks $v \in \text{adj\_in}[u]$ and $u \in \text{adj\_in}[v]$
    \item every edge with at least one endpoint in $A$ is captured, regardless of the cap
  \end{itemize}

  \vfill

  Cap limits frontier coverage (which papers can ever join $A \cup F$), not the correctness of $d_{\mathrm{frac}}(v) \cap A$.
\end{frame}

% ── A2. Reduced cost derivation (refs: Pricing a frontier node) ──────────────

\begin{frame}{A2.1. Reduced cost: setup}
  RMP at parameter $\lambda$, active set $A$:
  \begin{align*}
    \max \quad        & \sum_{(u,v)\in E(A)} y_{uv} - 2\lambda \sum_{\substack{u<v\\u,v\in A}} w_{uv} \\
    \text{s.t.} \quad & x_q = 1                                                                       \\
                      & \sum_{u\in A} x_u \ge k \quad (\text{dual: } \pi)                             \\
                      & y_{uv} \le x_u,\; y_{uv} \le x_v                                              \\
                      & w_{uv} \ge x_u + x_v - 1                                                      \\
                      & x,y,w \in [0,1]
  \end{align*}

  LP optimum: $\bar{x}, \bar{y}, \bar{w}$. Size dual: $\pi$.

  \vfill

  \textbf{Question.} If $f \in F$ joins $A$, does the LP improve?
\end{frame}

\begin{frame}{A2.2. Reduced cost: probe}
  Pretend $f$ joins $A$. New variables enter the LP:
  \begin{itemize}
    \item $x_f = 1$ (probe value)
    \item $y_{fv}$ for each edge $(f, v) \in E$ with $v \in A$
    \item $w_{fv}$ for each $v \in A$ (every pair, edge or not)
  \end{itemize}

  \vfill

  Freeze $\bar{x}_v$ for $v \in A$ at current LP values. Optimize new variables freely. The resulting objective change is the reduced cost:
  \[
    \bar{c}_f \;=\; \underbrace{(y\text{-gain})}_{\text{Step 1}} \;+\; \underbrace{(w\text{-penalty})}_{\text{Step 2}} \;+\; \underbrace{(\text{size correction})}_{\text{Step 3}}.
  \]
\end{frame}

\begin{frame}{A2.3. Reduced cost step 1: edge gain ($y$ terms)}
  For each new edge $(f, v) \in E$ with $v \in A$:

  \vfill

  \textbf{Bounds.}
  \[
    y_{fv} \le \min(x_f, x_v) = \min(1, \bar{x}_v) = \bar{x}_v.
  \]

  \textbf{Objective coefficient.} $+1$.

  \vfill

  Max problem, positive coefficient $\Rightarrow$ upper bound binds:
  \[
    y_{fv}^\star = \bar{x}_v.
  \]

  \vfill

  Total $y$-contribution:
  \[
    \sum_{v \in N(f) \cap A} \bar{x}_v \;=\; d_{\mathrm{frac}}(f).
  \]
\end{frame}

\begin{frame}{A2.4. Reduced cost step 2: pair penalty ($w$ terms)}
  For each $v \in A$ (every pair, edge or not):

  \vfill

  \textbf{Bound.}
  \[
    w_{fv} \ge x_f + x_v - 1 = \bar{x}_v.
  \]

  \textbf{Objective coefficient.} $-2\lambda$.

  \vfill

  Max problem, negative coefficient $\Rightarrow$ lower bound binds:
  \[
    w_{fv}^\star = \bar{x}_v.
  \]

  \vfill

  Total $w$-contribution:
  \[
    -2\lambda \sum_{v \in A} \bar{x}_v.
  \]
\end{frame}

\begin{frame}{A2.5. Reduced cost step 3: size dual ($\pi$ term)}
  Size constraint $\sum_{u \in A} x_u \ge k$ has dual price $\pi$.

  \vfill

  \textbf{Dual interpretation.} $\pi$ is the marginal LP value of relaxing $k$ by one unit.

  \vfill

  When $x_f = 1$, the LHS grows by one. The reduced cost picks up a shadow term $-\pi$.

  \vfill

  \begin{itemize}
    \item Size constraint slack at LP optimum: $\pi = 0$, no contribution.
    \item Size constraint binding ($|S| = k$): $\pi$ takes the sign that makes $-\pi$ the correct correction.
  \end{itemize}
\end{frame}

\begin{frame}{A2.6. Reduced cost: sum and decision}
  Sum the three contributions:
  \[
    \bar{c}_f \;=\; d_{\mathrm{frac}}(f) \;-\; 2\lambda \sum_{v \in A} \bar{x}_v \;-\; \pi.
  \]

  \vfill

  \textbf{Decision rule}
  \begin{itemize}
    \item $\bar{c}_f > 0$: adding $f$ strictly improves the LP $\Rightarrow$ promote to $A$.
    \item $\bar{c}_f \le 0$: no gain $\Rightarrow$ keep in $F$.
    \item $\max_{f \in F} \bar{c}_f \le 0$: no frontier node helps. Column generation terminates and the RMP LP objective is a valid upper bound on the parametric subproblem.
  \end{itemize}
\end{frame}

% ── A3. Branching threshold (refs: Branching rule) ───────────────────────────

\begin{frame}{A3. Branching threshold $2\lambda$}
  Dinkelbach objective:
  \[
    \sum_{(u,v)\in E} y_{uv} - 2\lambda \sum_{u<v} w_{uv}.
  \]
  Each pair carries penalty rate $2\lambda$.

  \vfill

  Marginal contribution of $v$ at fractional rate (per active neighbor $u$):
  \begin{itemize}
    \item edge gain: $+\bar{x}_u$ if $(u,v) \in E$
    \item pair penalty: $-2\lambda \bar{x}_u$ (every $u \in A$)
  \end{itemize}

  \vfill

  Aggregate: $v$ pays off when $d_{\mathrm{frac}}(v) \ge 2\lambda$.

  \vfill

  \begin{itemize}
    \item $d_{\mathrm{frac}}(v) \ge 2\lambda$: edge gain covers pair penalty rate $\Rightarrow$ core-like, branch one-first
    \item $d_{\mathrm{frac}}(v) < 2\lambda$: penalty dominates $\Rightarrow$ hanging, branch zero-first
  \end{itemize}

  \vfill

  Heuristic threshold; at LP optimum, included nodes typically satisfy $d_{\mathrm{frac}}(v) \ge 2\lambda$.
\end{frame}

% ── A4. $\kappa$-cut inequality (refs: $\kappa$-connectivity cuts) ───────────

\begin{frame}{A4. $\kappa$-cut inequality: derivation}
  \textbf{Menger's theorem (undirected).} For $t \in S$,
  \[
    \max(\text{edge-disjoint } q\text{-}t \text{ paths in } E(S)) \;=\; \min_{\substack{R \ni q \\ R \not\ni t}} |\delta(R) \cap E(S)|.
  \]
  So $S$ has $\ge \kappa$ edge-disjoint $q$-$t$ paths $\iff$ every $q$-$t$ cut $R$ has $\ge \kappa$ internal edges.

  \vfill

  \textbf{LP encoding.} $y_{uv} = 1$ iff edge $(u,v)$ internal to $S$.
  \begin{itemize}
    \item $x_t = 1$: need $\sum_{(u,v) \in \delta(R)} y_{uv} \ge \kappa$
    \item $x_t = 0$: no requirement
  \end{itemize}
  Combine:
  \[
    \sum_{(u,v) \in \delta(R)} y_{uv} \;\ge\; \kappa \, x_t.
  \]

  \vfill

  \textbf{Validity.} Holds at every $\kappa$-edge-connected integer $S$, so no feasible solution is excluded.
\end{frame}

% ── A5. LP formulation reference (refs: Dinkelbach transform, RMP) ───────────

\begin{frame}{A5.1. Nonlinear fractional 0-1 program}
  Let $x_v \in \{0,1\}$ indicate $v \in S$ and $y_{uv} \in \{0,1\}$ indicate edge $(u,v)$ is internal:
  \begin{align*}
    \max \quad          &
    \frac{\sum_{(u,v)\in E} y_{uv}}
    {\left(\sum_{u\in V}x_u\right)\left(\sum_{u\in V}x_u - 1\right)} \\
    \mathrm{s.t.} \quad & x_q = 1 \\
                        & \sum_{u\in V} x_u \ge k \\
                        & y_{uv} \le x_u,\quad y_{uv}\le x_v \quad \forall (u,v)\in E
  \end{align*}
  The lower bound $y_{uv}\ge x_u+x_v-1$ is not needed because $y$ has positive objective coefficient.
\end{frame}

\begin{frame}{A5.2. Dinkelbach parametric subproblem (bilinear form)}
  At parameter $\lambda$:
  \begin{align*}
    Z(\lambda) = \max \quad &
    \sum_{(u,v)\in E} y_{uv} - \lambda \left[ \left(\sum_{u\in V} x_u\right)^{\!2} - \sum_{u\in V} x_u \right] \\
    \mathrm{s.t.} \quad     &
    x_q = 1                                                                                                    \\
                            & \sum_{u\in V} x_u \ge k                                                          \\
                            & y_{uv} \le x_u,\quad y_{uv} \le x_v \quad \forall (u,v) \in E                    \\
                            & x \in \{0,1\}^{|V|},\ y \in \{0,1\}^{|E|}
  \end{align*}
\end{frame}

\begin{frame}{A5.3. Linearizing the penalty}
  $|S|(|S|-1) = (\sum_v x_v)^2 - \sum_v x_v$. For binary $x$, $x_v^2 = x_v$, so
  \[
    |S|(|S|-1) = 2\sum_{u<v} x_u x_v.
  \]
  Introduce pair variable $w_{uv} \approx x_u x_v$ with $w_{uv} \ge x_u + x_v - 1$. Since $w$ has negative coefficient, the lower bound is enough at integer solutions.
\end{frame}

\begin{frame}{A5.4. Linearized parametric subproblem}
  At parameter $\lambda$:
  \begin{align*}
    Z(\lambda) = \max \quad &
    \sum_{(u,v)\in E} y_{uv} - 2\lambda \sum_{u<v} w_{uv}                                                \\
    \mathrm{s.t.} \quad     &
    x_q = 1                                                                                              \\
                            & \sum_{u\in V} x_u \ge k                                                    \\
                            & y_{uv} \le x_u,\quad y_{uv} \le x_v \quad \forall (u,v) \in E              \\
                            & w_{uv} \ge x_u + x_v - 1 \quad \forall u<v                                 \\
                            & x \in \{0,1\}^{|V|},\ y \in \{0,1\}^{|E|},\ w \in \{0,1\}^{\binom{|V|}{2}}
  \end{align*}
\end{frame}

\begin{frame}{A5.5. Restricted Master Problem}
  For active set $A$:
  \begin{align*}
    \max \quad         &
    \sum_{(u,v)\in E(A)} y_{uv}
    -2\lambda\sum_{\substack{u,v\in A\\u<v}}w_{uv}         \\
    \mathrm{s.t.}\quad &
    x_q = 1                                                \\
                       & \sum_{u\in A}x_u \ge k
    \quad (\text{dual: }\pi)                               \\
                       & y_{uv}\le x_u,\quad y_{uv}\le x_v
    \quad \forall (u,v)\in E(A)                            \\
                       & w_{uv}\ge x_u+x_v-1
    \quad \forall u<v \in A                                \\
                       & 0\le x,y,w\le 1
  \end{align*}
\end{frame}

\begin{frame}{A5.6. Pruning integer incumbents}
  Since $|S| \ge k$, integer solutions may exceed $k$. Greedily remove nodes whose removal strictly improves the parametric objective.

  \vfill

  \begin{itemize}
    \item candidate: node with the smallest internal degree (weakest edge contribution)
    \item never remove $q$
    \item stop at $|S| = k$ or when no single removal helps
    \item preserve connectivity when $\kappa > 0$
  \end{itemize}

  \vfill

  \textbf{Effect.} Stronger incumbent, smaller returned community.
\end{frame}

\begin{frame}{A6. Exp 1 protocol detail}
  \textbf{Brute-force oracle}
  \begin{itemize}
    \item enumerate every $S \ni q$ with $|S| \ge k$
    \item for each $(k, \kappa)$, record the maximum-density $S$ with $\lambda(S) \ge \kappa$
    \item mark cells with no feasible $S$ as infeasible
  \end{itemize}

  \vfill

  \textbf{Solver}
  \begin{itemize}
    \item run on every (method, $k$, $\kappa$, seed) cell
    \item no time, node, gap, or Dinkelbach limit
    \item skip cells the brute force flagged infeasible (solver otherwise exhausts B\&B proving infeasibility; wall time blows up on sparse seeds)
  \end{itemize}
\end{frame}
